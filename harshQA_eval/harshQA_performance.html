def create_new_rank(row):
    i=row.name
    same_model=r1_dummies[(r1_dummies['Q_id']==row['Q_id'])&(r1_dummies['Company']==row['Company'])&(r1_dummies['Model']==row['Model'])&(r1_dummies['Domain']==row['Domain'])& (r1_dummies['Annoted_as_candidate']==1)]
    other_model=r1_dummies[(r1_dummies['Q_id']==row['Q_id'])&(r1_dummies['Company']==row['Company'])&(r1_dummies['Model']!=row['Model'])&(r1_dummies['Domain']==row['Domain'])& (r1_dummies['Annoted_as_candidate']==1)]
    if len(same_model)==0:
        if len(other_model)==0:
            row['rank_first_candidate']=0
        else:
            row['rank_first_candidate']=100
    else:
        rank=same_model.iloc[0]['Rank']
        row['rank_first_candidate']=int(rank)
    return row

r1_dummies=r1_dummies.apply(create_new_rank,axis=1)

def func_AP(row):
    i=row.name
    
    m,qid,c=row.Model,row.Q_id,row.Company
    same=r1_dummies[(r1_dummies.Model==m )& (r1_dummies.Company==c) & (r1_dummies.Q_id==qid)]
    count=0
    count_cum=0

    
    for k in range(5):
        label=same.iloc[k]['Annoted_as_candidate']
        if label==1:
            count+=1
            count_cum+=count/(k+1)
    if count==0:
        return 0
    else:
        return count_cum/count
    

r1_dummies['AP']=r1_dummies.apply(func_AP,axis=1)
